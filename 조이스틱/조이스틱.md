def dfs(now, count, name, n, notA):
    if not notA: return count
    # ì˜¤ë¥¸ìª½ìœ¼ë¡œ ì´ë™

def solution(name):
    answer = 0
    notA = []
    n = len(name)
    for i in range(n):
        if name[i] != "A":
            notA.append(i)
            answer += min(ord(name[i])-ord("A"), ord("Z")-ord(name[i])+1)
    answer += dfs(0, 0, name, n, notA)
    return answer
