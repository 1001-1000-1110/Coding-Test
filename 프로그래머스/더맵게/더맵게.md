```java
import java.util.*;
class Solution {
    public int solution(int[] scoville, int K) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for(int val : scoville){
            pq.add(val);
        }
        int temp = pq.peek(); //í˜„ì¬ ê°€ì¥ ë‚®ì€ ìŠ¤ì½”ë¹Œ ì§€ìˆ˜
        int cnt = 0; //ëª‡ë²ˆ ê³„ì‚° í–ˆëŠ”ì§€
        while(temp < K && pq.size() >= 2){
            int first = pq.poll();
            int second = pq.poll();
            pq.add(first + second * 2);
            temp = pq.peek();
            cnt++;
        }
        
        if(temp < K) return -1;
        return cnt;
    }
}

```
