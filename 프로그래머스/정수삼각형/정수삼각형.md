class Solution {
    public int solution(int[][] triangle) {
        int n = triangle[triangle.length - 1].length;

        for(int i = 1; i < triangle.length; i++){
            for(int j = 0; j < triangle[i].length; j++){
                if(j == 0){ //í–‰ì˜ ì²«ë²ˆì§¸ ê°’ì¸ ê²½ìš°, ì¢Œì¸¡ ìƒë‹¨ì˜ ê°’ì´ ì¡´ì¬í•˜ì§€ ì•ŠìŒ.
                    triangle[i][j] = triangle[i - 1][j] + triangle[i][j];
                    continue;
                }
                if(j == triangle[i].length - 1){ //í–‰ì˜ ë§ˆì§€ë§‰ ê°’ì¸ ê²½ìš°, ìš°ì¸¡ ìƒë‹¨ì˜ ê°’ì´ ì¡´ì¬í•˜ì§€ ì•ŠìŒ.
                    triangle[i][j] = triangle[i - 1][j - 1] + triangle[i][j];
                    continue;
                }
                triangle[i][j] = Math.max(triangle[i - 1][j - 1], triangle[i - 1][j]) + triangle[i][j];
            }
        }


        int answer = 0;
        for(int i = 0; i < n; i++){
            answer = Math.max(triangle[triangle.length - 1][i], answer);
        }
        return answer;
    }
}
