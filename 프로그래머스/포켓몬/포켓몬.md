```java
import java.util.*;
class Solution {
    public int solution(int[] nums) {

        Set<Integer> set = new HashSet<Integer>();
        for(int val : nums){
            set.add(val);
        }
        if(nums.length/2 <= set.size())
            return nums.length/2;

        return set.size();

    }
}
```
### Time Complexity : O(N)
