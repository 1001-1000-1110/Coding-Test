import java.io.*;

        int[] nums = new int[N + 1];

    private static void fillIsPalindrome(int N, int[] nums) {
        for(int i = 1; i <= N; i++){
            for(int j = 1; j <= i ; j++){
                int left = j;
                int right = i;
