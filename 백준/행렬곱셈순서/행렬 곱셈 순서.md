```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int N = Integer.parseInt(br.readLine());
        int[][] matrix = new int[N + 1][2];
        int[][] dp = new int[N + 1][N + 1];

        //matrix
        for(int i = 1;  i <= N; i++){
            StringTokenizer st = new StringTokenizer(br.readLine());
            matrix[i][0] = Integer.parseInt(st.nextToken());
            matrix[i][1] = Integer.parseInt(st.nextToken());
        }

        //ìµœì†Œ ê³„ì‚° íšŸìˆ˜ë¥¼ êµ¬í•¨
        for(int i = 1;  i <= N; i++){ //êµ¬ê°„ì˜ ê°„ê²©
            for(int j = 1; j <= N - i; j++){ //êµ¬ê°„ì˜ ì‹œì‘ì 
                dp[j][j + i] = Integer.MAX_VALUE;
                for(int k = j; k < j + i; k++){ //êµ¬ê°„ì„ ìª¼ê°œëŠ” ì§€ì 
                    dp[j][i + j] = Math.min(dp[j][j + i], dp[j][k] + dp[k + 1][j + i] + (matrix[j][0] * matrix[k + 1][0] * matrix[j + i][1]));
                }
            }
        }

        System.out.println(dp[1][N]);
    }
}

```
