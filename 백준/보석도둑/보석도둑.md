import java.io.BufferedReader;
import java.io.IOException;

import java.util.*;

public class Main {
    static class Jewel implements Comparable<Jewel>{

        int price;
        public Jewel(int weight, int price){

    public static void main(String[] args) throws IOException {

        int price = 0;
        //ê°€ê²©ë³„ë¡œ ì •ë ¬í•¨.

        for(int i = 0; i < n; i++){

        long answer = 0;
        PriorityQueue<Integer> pq = new PriorityQueue<>(Collections.reverseOrder()); //ìµœëŒ€í™, ê°€ì¥ í° ê°€ê²©ì„ ë°˜í™˜í•¨.

            for(int j = index; j < jewels.size(); j++){

                    pq.offer(jewels.get(j).price);//ê°€ë°©ì— ë‹´ì„ ìˆ˜ ìˆëŠ” ë³´ì„ë“¤ì˜ ê°€ê²©ì„ ìš°ì„ ìˆœìœ„ í(ìµœëŒ€ í™)ì— ë„£ëŠ”ë‹¤.

        }
