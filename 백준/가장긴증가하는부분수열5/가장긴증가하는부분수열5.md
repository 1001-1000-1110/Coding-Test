import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.StringTokenizer;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int n = Integer.parseInt(br.readLine());

        int[] nums = new int[n];
        StringTokenizer st = new StringTokenizer(br.readLine());
        for (int i = 0; i < n; i++) {
            nums[i] = Integer.parseInt(st.nextToken());
        }

        int[] arr = new int[n];
        arr[0] = nums[0];

        int[] index = new int[n];//index[a] = b ì›ë³¸ë°ì´í„°ì—ì„œ aë²ˆì§¸ì— ìˆëŠ” ìˆ«ìëŠ” ,arrë°°ì—´ì— bì— ìœ„ì¹˜íŒë‹¤.
        index[0] = 0;

        int lastIndex = 0;
        for (int i = 1; i < n; i++) {
            int now = nums[i];
            if (now > arr[lastIndex]) { //nowê°€ arrì˜ ê°€ì¥ ëì˜ ìˆ«ìë³´ë‹¤ í°ê²½ìš°
                lastIndex++;
                arr[lastIndex] = now;
                index[i] = lastIndex;
            } else {//ì‘ê±°ë‚˜ ê°™ì€ ê²½ìš°
                //arrì˜ ë°°ì—´ì—ì„œ ìì‹ ë³´ë‹¤ í° ê²ƒë“¤ì¤‘ ê°€ì¥ ì‘ì€ ê°’ê³¼ êµì²´í•¨
                int left = 0;
                int right = lastIndex;
                while (left < right) {
                    int mid = (left + right) / 2;
                    if (arr[mid] < now) {
                        left = mid + 1;
                    } else {
                        right = mid;
                    }
                }
                arr[left] = now;
                index[i] = left;
            }

        }

        //ê¸¸ì´ ì¶œë ¥
        int len = lastIndex + 1;
        System.out.println(len);

        //ë°°ì—´ ì¶œë ¥, indexë°°ì—´ì„ ê±°ê¾¸ë¡œ ìˆœíšŒí•˜ë©´ì„œ lastIndexì™€ ë™ì¼í•œ ê°’ì„ ê°€ì§„ ê°’ì„ resultë°°ì—´ì— ë„£ìŒ
        int[] result = new int[len];
        for (int i = index.length - 1; i >= 0; i--) {
            if (index[i] == lastIndex) {
                result[lastIndex] = nums[i];
                lastIndex--;
            }
            if (lastIndex < 0) {
                break;
            }
        }
        //resultë°°ì—´ ì¶œë ¥
        Arrays.stream(result).forEach(num -> System.out.print(num + " "));
    }
}
