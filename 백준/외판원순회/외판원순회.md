import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

public class Main {
    static int[][] dp;//dp[í˜„ì¬ ë…¸ë“œ][í˜„ì¬ ë…¸ë“œë¥¼ í¬í•¨, ë°©ë¬¸í•œ ë…¸ë“œë¥¼ ë¹„íŠ¸ë§ˆìŠ¤í¬ë¡œ ë‚˜íƒ€ëƒ„]
    static int[][] map;
    static int N;
    static int visitedAll;//ëª¨ë‘ ë°©ë¬¸í–ˆì„ ê²½ìš°ì˜ ë¹„íŠ¸
    static int INF = 16_000_000;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        N = Integer.parseInt(br.readLine());

        map = new int[N + 1][N + 1];
        //ì¸ì ‘ë¦¬ìŠ¤íŠ¸ ì´ˆê¸°í™”;
        for (int i = 1; i <= N ; i++) {
            StringTokenizer st = new StringTokenizer(br.readLine());
            for(int j = 1; j <= N;  j++){
                map[i][j] = Integer.parseInt(st.nextToken());
            }
        }

        visitedAll = (1 << N) - 1;
        dp = new int[N + 1][visitedAll + 1];

        for(int i = 1 ; i <= N ; i++) {
            for(int j = 1 ; j <= visitedAll ; j++) {
                dp[i][j] = -1;
            }
        }

        int result = tsp(1,1);
        System.out.println(result);
    }

    private static int tsp(int now, int visited) {

        if(visited == visitedAll){ //ëª¨ë“  ë„ì‹œë¥¼ ë°©ë¬¸í•œ ê²½ìš°
            if(map[now][1] == 0){
                return INF; //ê²½ë¡œê°€ ì—†ìœ¼ë©´ INFë¥¼ ë°˜í™˜í•œë‹¤.(í˜¸ì¶œí•œ ê³³ì—ì„œ Math.minì„ í†µí•´ ìµœì†Œê°’ì„ ì €ì¥í•˜ê¸° ë•Œë¬¸)
            }else{
                return map[now][1]; //ì‹œì‘ ë„ì‹œì¸ 1ë¡œ ê°€ëŠ” ë¹„ìš©ì„ ë°˜í™˜í•¨.
            }
        }

        //ì´ë¯¸ ë°©ë¬¸í–ˆë˜ ê²½ìš°, ê¸°ì¡´ ê°’ì„ ë°˜í™˜í•œë‹¤.
        if(dp[now][visited] != -1)
            return dp[now][visited];
        
        //ë°©ë¬¸í–ˆìŒì„ ë‚˜íƒ€ë‚¸ë‹¤.
        dp[now][visited] = INF;
        //ë‹¤ìŒì— ë°©ë¬¸í•  ìˆ˜ ìˆëŠ” êµ¬ì—­ì´ ì¡´ì¬í•˜ë©´, ë°©ë¬¸í•¨
        for(int i = 1; i <= N; i++){
            if(map[now][i] == 0)//ê²½ë¡œê°€ ì—†ëŠ” ê²½ìš°
                continue;
            int next = (1 << (i - 1));
            if((visited & next) != 0) //ì´ë¯¸ ë°©ë¬¸í•œ ê²½ìš°
                continue;
            //ë‹¤ìŒ êµ¬ê°„ì— ë°©ë¬¸í•  ìˆ˜ ìˆëŠ” ê²½ìš°
            dp[now][visited] = Math.min(dp[now][visited], tsp(i,visited | next) + map[now][i]);
        }
        return dp[now][visited];
    }
}
