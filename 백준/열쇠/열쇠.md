import java.io.*;
import java.util.*;

public class Main {
    static boolean[] Keys; //í‚¤ë¥¼ ê°€ì§€ê³  ìˆëŠ”ì§€ í™•ì¸í•˜ê¸° ìœ„í•œ ë°°ì—´
    static int h, w; //mapì˜ í¬ê¸°

    static char[][] Map;

    static int cnt; //íšë“í•œ ë¬¸ì„œì˜ ê°œìˆ˜
    private static class Point { //ì¢Œí‘œë¥¼ ê¸°ë¡í•˜ê¸° ìœ„í•œ í´ë˜ìŠ¤
        int x;
        int y;

        public Point(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));

        int t = Integer.parseInt(br.readLine());

        for (int i = 0; i < t; i++) { //í…ŒìŠ¤í¬ ì¼€ì´ìŠ¤ ë§Œí¼ ë°˜ë³µ
            bw.write(solve(br) + "\n");
        }

        bw.flush();
        bw.close();
    }

    private static int solve(BufferedReader br) throws IOException {
        //ì´ˆê¸°í™”
        StringTokenizer st = new StringTokenizer(br.readLine());
        h = Integer.parseInt(st.nextToken());
        w = Integer.parseInt(st.nextToken());
        visited = new boolean[h][w];
        cnt = 0;
        Map = new char[h][w];
        //ì…ë ¥ ë°›ì€ ê°’ìœ¼ë¡œ mapì„ ë§Œë“¬.
        for (int i = 0; i < h; i++) {
            Map[i] = br.readLine().toCharArray();
        }
        Keys = new boolean[26];
        DoorPoints = new ArrayList[26];
        for (int i = 0; i < DoorPoints.length; i++) {
            DoorPoints[i] = new ArrayList<>();
        }
        //ì…ë ¥ìœ¼ë¡œ ì£¼ì–´ì§„ í‚¤ë¥¼ ì…ë ¥ ë°›ìŒ.
        String keyStr = br.readLine();

        if (!keyStr.equals("0")) {//í‚¤ë¥¼ ê°€ì§€ê³  ìˆëŠ” ê²½ìš°
            for (int i = 0; i < keyStr.length(); i++) {
                Keys[keyStr.charAt(i) - 'a'] = true;
            }
        }

        //ì‹œì‘ ì§€ì ì„ ì°¾ì•„ì„œ bfsë¥¼ ì§„í–‰í•¨.
        //ì‹œì‘ì§€ì  = ë²½ì´ ì•„ë‹Œ ê³³
        findStartingPoint();
        return cnt;
    }

    private static void findStartingPoint() {
        for (int i = 0; i < h; i++) { //ë¹Œë”©ì˜ ê°€ì¥ ìœ„, ì•„ë˜ ê°€ë¡œë°©í–¥
            for (int j = 0; j < w; j++) {
                if (i > 0 && i < h - 1 && j > 0 && j < w - 1) {
                    continue;
                }
                if(visited[i][j]){
                    continue;
                }

                if (Map[i][j] != '*') {
                    bfs(i, j);
                }
            }
        }
    }

    private static void bfs(int x, int y) {
        Queue<Point> q = new LinkedList<>();
        q.add(new Point(x, y));
//        System.out.println("startPoint x: "+x+" y: "+y);

        visited[x][y] = true;
        //ì‹œì‘ ìœ„ì¹˜ê°€ ë¬¸ì¸ ê²½ìš°
        if(Map[x][y] >= 'A' &&  Map[x][y] <= 'Z'){
            if(!Keys[Map[x][y] - 'A']){ //í‚¤ë¥¼ ê°€ì§€ê³  ìˆì§€ ì•Šì€ ê²½ìš°, ë”ì´ìƒ ì§„ì…ì´ ë¶ˆê°€ëŠ¥
                DoorPoints[Map[x][y] - 'A'].add(new Point(x, y));//ë¬¸ì˜ ìœ„ì¹˜ë¥¼ ê¸°ë¡í•¨
                return;//bfsë©”ì„œë“œë¥¼ ì¢…ë£Œí•¨
            }
        }

        //ì‹œì‘ ìœ„ì¹˜ê°€ í‚¤ì¸ ê²½ìš°
        if (Map[x][y] >= 'a' && Map[x][y] <= 'z') {
            Keys[Map[x][y] - 'a'] = true; //í‚¤ë¥¼ ê°€ì§€ê²Œ ë˜ì—ˆìœ¼ë¯€ë¡œ trueì²˜ë¦¬
        }

        //ì‹œì‘ ìœ„ì¹˜ê°€ ë¬¸ì„œì¸ ê²½ìš°
        if (Map[x][y] == '$') {
            cnt++;
        }

        while (!q.isEmpty()) {
            Point now = q.poll();
            for (int i = 0; i < 4; i++) {
                int nx = now.x + dx[i];
                int ny = now.y + dy[i];

                //ì£¼ì–´ì§„ ë§µì„ ë²—ì–´ë‚˜ì§€ ì•Šì€ ê²½ìš°, í•œë²ˆë„ ë°©ë¬¸í•˜ì§€ ì•Šì€ ê²½ìš°, ë²½ì´ ì•„ë‹Œ ê²½ìš°
                if (nx >= 0 && nx < h && ny >= 0 && ny < w && !visited[nx][ny] && Map[nx][ny] != '*') {
                    visited[nx][ny] = true;
                    //ë¬¸ì¸ ê²½ìš°
                    if (Map[nx][ny] >= 'A' && Map[nx][ny] <= 'Z') {
                        if (!Keys[Map[nx][ny] - 'A']) { //í‚¤ê°€ ì—†ëŠ” ê²½ìš°
                            DoorPoints[Map[nx][ny] - 'A'].add(new Point(nx, ny)); //í˜„ì¬ ì—´ìˆ˜ ì—†ëŠ” ë¬¸ì˜ ì¢Œí‘œë¥¼ ì €ì¥í•¨.
                            continue;
                        }
                    }
                    //í‚¤ì¸ ê²½ìš°
                    else if (Map[nx][ny] >= 'a' && Map[nx][ny] <= 'z') {
                        if (!Keys[Map[nx][ny] - 'a']) { //í‚¤ë¥¼ ê°€ì§€ê³  ìˆì§€ ì•Šì€ ê²½ìš°
                            Keys[Map[nx][ny] - 'a'] = true;
                            //íì— íšë“í•œ í‚¤ë¡œ ë°©ë¬¸í•  ìˆ˜ ìˆëŠ” ê³³ì„ ì¶”ê°€í•¨.
                            for(Point point : DoorPoints[Map[nx][ny] - 'a']){
                                q.add(point);
                            }
                        }
                    }
                    //ë¬¸ì„œì¸ ê²½ìš°
                    else if (Map[nx][ny] == '$') {
                      cnt++;
                    }

                    q.add(new Point(nx, ny));
                }
            }
        }
    }


}
