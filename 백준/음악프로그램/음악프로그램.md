import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        StringTokenizer st = new StringTokenizer(br.readLine());
        int n = Integer.parseInt(st.nextToken()); //ê°€ìˆ˜
        int m = Integer.parseInt(st.nextToken()); //pd

        int[] indegree = new int[n + 1];
        List<Integer>[] map = new ArrayList[n + 1];
        for(int i = 0; i <= n; i++){
            map[i] = new ArrayList<>();
        }
        //ê° pdë“¤ì´ ì œì‘í•œ ìˆœì„œë¥¼ ê¸°ì¤€ìœ¼ë¡œ mapì„ ì´ˆê¸°í™”í•¨
        for (int i = 0; i < m; i++) {

            for(int j = 2; j < singers.length; j++){
                indegree[singers[j]]++; //ì§„ì…ì°¨ìˆ˜
                map[singers[j - 1]].add(singers[j]); //map

        Queue<Integer> q = new LinkedList<>();

                q.add(i);
                bw.write(i+"\n");
            }
        }
        while(!q.isEmpty()){
            int now = q.poll();
            //í˜„ì¬ ë…¸ë“œ(ê°€ìˆ˜) ë‹¤ìŒì— ì—°ê²°ëœ ë…¸ë“œì˜ ì§„ì…ì°¨ìˆ˜ë¥¼ -1í•¨
            for (Integer next : map[now]) {

                if(indegree[next] == 0){
                    q.add(next);
                    bw.write(next+"\n");
                }
            }
        }

        for(int i = 1; i <= n; i++){
            if(indegree[i] != 0){ //ìˆœì„œ ì •í•˜ëŠ” ê²ƒì´ ë¶ˆê°€ëŠ¥í•œ ê²½ìš°
                System.out.println(0);

    }
}
