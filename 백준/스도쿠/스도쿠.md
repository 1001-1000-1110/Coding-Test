import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.ArrayList;
import java.util.List;

public class Main {
    static int N = 9;
    static int[][] map = new int[N][N];
    static List<Point> zeroPoints = new ArrayList<>(); //ì±„ì›Œì•¼ í•˜ëŠ” ì¢Œí‘œë¥¼ ì €ì¥í•¨.
    static boolean isEnd;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        //9*9ë§µ í•„ìš”
        for (int i = 0; i < N; i++) {
            String str = br.readLine();
            for (int j = 0; j < N; j++) {
                map[i][j] = str.charAt(j) - '0';
                if (map[i][j] == 0) { //0ì¸ê²½ìš° ìˆ«ìë¥¼ ì±„ì›Œì•¼í•˜ëŠ” pointì´ë¯€ë¡œ ì €ì¥
                    zeroPoints.add(new Point(i, j));
                }
            }
        }
        dfs(0);
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                System.out.print(map[i][j]);
            }
            System.out.println();
        }

    }

    private static void dfs(int index) {
        if (index == zeroPoints.size()) {
            isEnd = true;  //ìŠ¤ë„ì¿ ê°€ ì²˜ìŒ ì™„ì„±ë˜ë©´ ë”ì´ìƒ ì¬ê·€ë¥¼ ë°˜ë³µí•˜ì§€ ì•Šê³  ì¢…ë£Œí•¨.
            return;
        }
        Point now = zeroPoints.get(index);
        for (int i = 1; i <= N; i++) {
            map[now.x][now.y] = i;
            if (checkCol(now.x, now.y) && checkRow(now.x, now.y) && checkSquare(now.x, now.y)) {
                dfs(index + 1);
            }
            if (isEnd) { //ìŠ¤ë„ì¿ ê°€ ì²˜ìŒ ì™„ì„±ë˜ë©´ ë”ì´ìƒ ì¬ê·€ë¥¼ ë°˜ë³µí•˜ì§€ ì•Šê³  ì¢…ë£Œí•¨.
                return;
            }
            map[now.x][now.y] = 0;
        }
    }

    //[][][]
    //[][][]
    //[][][] ì²˜ëŸ¼ ì¢Œí‘œ(point)ê°€ í¬í•¨ëœ 3*3 í¬ê¸°ì˜ êµ¬ê°„ì— ì¤‘ë³µë˜ëŠ” ê°’ì´ ì—†ëŠ”ì§€ í™•ì¸í•¨.
    private static boolean checkSquare(int x, int y) {
        //ê° ì¢Œí‘œì˜ ì‘ì€ 3*3 ì‚¬ê°í˜• êµ¬ê°„ì˜ ì‹œì‘ì§€ì ì„ ì •í•¨.
        int startX = (x / 3) * 3;
        int startY = (y / 3) * 3;

        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                if (startX + i == x && startY + j == y) continue;
                if (map[startX + i][startY + j] == map[x][y]) {
                    return false;
                }
            }
        }
        return true;
    }

    //ê°€ë¡œì— ë™ì¼í•œ ê°’ì´ ìˆëŠ”ì§€ í™•ì¸
    private static boolean checkRow(int x, int y) {
        for (int i = 0; i < N; i++) {
            if (i == y)
                continue;
            if (map[x][y] == map[x][i]) {
                return false;
            }
        }
        return true;
    }

    //ì„¸ë¡œì— ë™ì¼í•œ ê°’ì´ ìˆëŠ”ì§€ í™•ì¸
    private static boolean checkCol(int x, int y) {
        for (int i = 0; i < N; i++) {
            if (i == x)
                continue;
            if (map[x][y] == map[i][y]) {
                return false;
            }
        }
        return true;
    }


    private static class Point {
        int x;
        int y;

        public Point(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }
}
